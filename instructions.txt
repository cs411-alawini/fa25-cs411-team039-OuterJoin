Hosting (in car_tinder)

PROJECT=car-tinder-476522
REGION=us-central1
REPO=web-apis
IMAGE="us-central1-docker.pkg.dev/$PROJECT/$REPO/car-tinder-api"
INSTANCE="car-tinder-476522:us-central1:car-tinder-db-1"

# rebuild/push
gcloud builds submit server --tag "$IMAGE"

# redeploy (note: no PORT= env)
gcloud run deploy car-tinder-api \
  --image "$IMAGE" \
  --region "$REGION" \
  --platform managed \
  --allow-unauthenticated \
  --service-account "car-tinder-api-sa@$PROJECT.iam.gserviceaccount.com" \
  --add-cloudsql-instances "$INSTANCE" \
  --set-env-vars "INSTANCE_CONNECTION_NAME=$INSTANCE,DB_USER=dev_user,DB_NAME=car_tinder" \
  --set-secrets "DB_PASSWORD=car-tinder-db-password:latest"

(in client directory)
npm run build
